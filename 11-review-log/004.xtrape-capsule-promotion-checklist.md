
必须包含一个最小示例：

```ts
import { createCapsuleAgent } from '@xtrape/capsule-agent-node';

const agent = createCapsuleAgent({
  opstageUrl: process.env.OPSTAGE_URL!,
  registrationToken: process.env.OPSTAGE_REGISTRATION_TOKEN!,
  service: {
    code: 'hello-capsule',
    name: 'Hello Capsule',
    version: '0.1.0',
    runtime: 'nodejs'
  },
  health: async () => ({
    status: 'UP',
    details: {
      uptime: process.uptime()
    }
  }),
  actions: [
    {
      name: 'echo',
      label: 'Echo',
      prepare: async () => ({
        schema: {
          type: 'object',
          properties: {
            message: { type: 'string' }
          }
        }
      }),
      execute: async (payload) => ({
        result: {
          echo: payload.message
        }
      })
    }
  ]
});

await agent.start();
```

注意：示例 API 名称要按实际 SDK 调整。
